/*global MapInformation,Point,Bounds,LatLng,Rectangle,Publisher,Events*/
 
describe('MapInformation', function() {
    "use strict";
 
    let id,
        info,
        eventManager;
 
    beforeEach(() => {
        id = Date.now();
        info = new MapInformation.MapInformation(id);
        eventManager = new Publisher.Publisher(id);
 
        const nw = new LatLng.LatLng(90, -180);
        const se = new LatLng.LatLng(-90, 180);
 
        eventManager.publish(Events.Events.MapInformation.UPDATE, {
            zoomFactor: 1,
            bounds: new Bounds.Bounds(nw, se),
            view: new Rectangle.Rectangle(-1800, 900, 3600, 1800),
            viewport: new Rectangle.Rectangle(0, 0, 360, 180)
        });
     });
 
    it("is an instanceof MapInformation", function() {
        expect(info instanceof MapInformation.MapInformation).toBeTrue;
    });
 
    it("pixelPerLatLng equals is correct", function() {
        expect(info.pixelPerLatLng).toEqual(new Point.Point(10,10));
    });
 
    it("convertPointToLatLng", function() {
        const latlng1 = info.convertPointToLatLng(new Point.Point( -1800, 900));
        const latlng2 = info.convertPointToLatLng(new Point.Point(1800, 900));
        const latlng3 = info.convertPointToLatLng(new Point.Point(-1800, -900));
        const latlng4 = info.convertPointToLatLng(new Point.Point(1800, -900));
        const latlng5 = info.convertPointToLatLng(new Point.Point(0, 0));
        expect(latlng1).toEqual(new LatLng.LatLng(90, -180));
        expect(latlng2).toEqual(new LatLng.LatLng(90, 180));
        expect(latlng3).toEqual(new LatLng.LatLng(-90, -180));
        expect(latlng4).toEqual(new LatLng.LatLng(-90, 180));
        expect(latlng5).toEqual(new LatLng.LatLng(0, 0));
    });
 
    it("convertLatLngToPoint", function() {
        const p1 = info.convertLatLngToPoint(new LatLng.LatLng(90, -180));
        const p2 = info.convertLatLngToPoint(new LatLng.LatLng(90, 180));
        const p3 = info.convertLatLngToPoint(new LatLng.LatLng(-90, -180));
        const p4 = info.convertLatLngToPoint(new LatLng.LatLng(-90, 180));
        const p5 = info.convertLatLngToPoint(new LatLng.LatLng(0, 0));
        expect(p1).toEqual(new Point.Point( -1800, 900));
        expect(p2).toEqual(new Point.Point(1800, 900));
        expect(p3).toEqual(new Point.Point(-1800, -900));
        expect(p4).toEqual(new Point.Point(1800, -900));
        expect(p5).toEqual(new Point.Point(0, 0));
    });
 
 
    afterEach(() => {
        eventManager.destroy();
        info = undefined;
     });
 
 
 
});
