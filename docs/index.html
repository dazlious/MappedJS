new Interact(settings)

Constructor

Returns: Interact - new instance
Params

  • settings Object - = {} - all the settings
    • .container string | Object - = ".interact-container" - Container, either string, jQuery-object or dom-object
    • .timeTreshold Object - = {} - settings for the timing tresholds
      • .tap number - = 200 - timing treshold for tap
      • .hold number - = 500 - timing treshold for hold
      • .swipe number - = 300 - timing treshold for swipe
      • .flick number - = 30 - timing treshold for flick
    • .distanceTreshold Object - = {} - settings for the distance tresholds
      • .swipe number - = 200 - distance treshold for swipe
    • .overwriteViewportSettings Boolean | string - = false - on true prevents pinching, can be a custom string too
    • .stopPropagation Boolean - = true - on true stops the propagation of events
    • .preventDefault Boolean - = true - on true prevents the default actions of events
    • .autoFireHold Boolean - = false - if set to false hold-event is not fired
    • .pinchBalanceTime number - = 50 - prevents from firing too much pinching events
    • .callbacks Object - = {} - settings for the callback-functions
      • .tap function - = null - callback-function for tap
      • .tapHold function - = null - callback-function for tapHold
      • .doubletap function - = null - callback-function for doubletap
      • .hold function - = null - callback-function for hold
      • .pan function - = null - callback-function for pan
      • .swipe function - = null - callback-function for swipe
      • .flick function - = null - callback-function for flick
      • .zoom function - = null - callback-function for zoom
      • .wheel function - = null - callback-function for wheel
      • .pinch function - = null - callback-function for pinch
    • .events Object - = {} - settings all eventnames
      • .start Object - = {} - settings all start eventnames
        • .touch Object - = ("MSPointerDown pointerdown" || "touchstart") - settings start touch eventnames
        • .mouse Object - = ("MSPointerDown pointerdown" || "mousedown") - settings start mouse eventnames
      • .move Object - = {} - settings all move eventnames
        • .touch Object - = ("MSPointerMove pointermove" || "touchmove") - settings move touch eventnames
        • .mouse Object - = ("MSPointerMove pointermove" || "mousemove") - settings move mouse eventnames
      • .end Object - = {} - settings all end eventnames
        • .touch Object - = ("MSPointerUp pointerup" || "touchend") - settings end touch eventnames
        • .mouse Object - = ("MSPointerUp pointerup" || "mouseup") - settings end mouse eventnames
      • .leave Object - = {} - settings all leave eventnames
        • .touch Object - = ("MSPointerLeave pointerleave" || "touchleave") - settings leave touch eventnames
        • .mouse Object - = ("MSPointerLeave pointerleave" || "mouseleave") - settings leave mouse eventnames
      • .scroll string - = ("wheel" || "mousewhell" || "DOMMouseScroll") - settings all scroll eventnames
